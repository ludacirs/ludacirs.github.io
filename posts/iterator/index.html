const range = {
  from: 1,
  to: 5,
};

range[Symbol.iterator] = function (){
  return {
    current: this.from,
    last: this.to,

    next(){
      if(this.current <= this.last){
        return {done: false, value: this.current++};
      } else{
        return {done: true};
      }
    }
  }
}

for(let v of range){
  console.log(v); // 1, 2, 3, 4, 5
}
